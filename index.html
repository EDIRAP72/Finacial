// ========================= // HEATMAP // ========================= const heatmapGrid = document.getElementById("heatmap-grid"); const heatmapView = document.getElementById("heatmap-view"); const detailView = document.getElementById("detail-view"); const detailTitle = document.getElementById("detail-title"); const detailStats = document.getElementById("detail-stats"); const entriesTableBody = document.querySelector("#entries-table tbody"); const backBtn = document.getElementById("back-btn"); const chartCanvas = document.getElementById("chart-canvas"); const ctx = chartCanvas.getContext("2d"); function moneyFlowToColor(mf) { // mf 0..1 -> da rosso (scarico) a verde (carico) const r = Math.round(255 * (1 - mf)); const g = Math.round(255 * mf); const b = 60; return `rgb(${r},${g},${b})`; } function renderHeatmap() { heatmapGrid.innerHTML = ""; instruments.forEach((inst, idx) => { const tile = document.createElement("div"); tile.className = "tile"; tile.style.background = `linear-gradient(135deg, ${moneyFlowToColor(inst.moneyFlow)} 0%, #05070a 60%)`; const ticker = document.createElement("div"); ticker.className = "ticker"; ticker.textContent = inst.ticker; const type = document.createElement("div"); type.className = "asset-type"; type.textContent = inst.type; const metric1 = document.createElement("div"); metric1.className = "metric"; metric1.innerHTML = `Prezzo: ${inst.basePrice.toFixed(2)}`; const metric2 = document.createElement("div"); metric2.className = "metric"; const score = (inst.moneyFlow * inst.liquidity * 1_000_000).toFixed(0); metric2.innerHTML = `Score flusso: ${score}`; const metric3 = document.createElement("div"); metric3.className = "metric"; const volRel = (0.8 + inst.moneyFlow * 1.4).toFixed(2); metric3.innerHTML = `Vol rel: ${volRel}`; const badge = document.createElement("div"); badge.className = "badge"; badge.textContent = "Domanda/Offerta attiva"; tile.appendChild(ticker); tile.appendChild(type); tile.appendChild(metric1); tile.appendChild(metric2); tile.appendChild(metric3); tile.appendChild(badge); tile.addEventListener("click", () => openDetail(inst, idx)); heatmapGrid.appendChild(tile); }); } // ========================= // DETTAGLIO: CALCOLO + GRAFICO // ========================= function generateHistoricalSeries(basePrice, moneyFlow, liquidity) { // 3 anni ~ 36 mesi (semplificato) const points = 36; const data = []; let price = basePrice * (0.7 + Math.random() * 0.2); // parte un po' più in basso const trend = (moneyFlow - 0.5) * 0.08 * basePrice; // trend medio annuo const monthlyTrend = trend / 12; for (let i = points - 1; i >= 0; i--) { const noise = (Math.random() - 0.5) * basePrice * 0.03 * (1 + (1 - liquidity)); price = Math.max(0.1, price + monthlyTrend + noise); data.unshift(price); } return data; } function projectFutureSeries(lastPrice, moneyFlow, liquidity) { // Proiezione 1 mese ~ 20 punti (giorni) const points = 20; const data = []; let price = lastPrice; const dailyTrend = (moneyFlow - 0.5) * 0.02 * lastPrice; // trend giornaliero sintetico for (let i = 0; i < points; i++) { const noise = (Math.random() - 0.5) * lastPrice * 0.01 * (1 + (1 - liquidity)); price = Math.max(0.1, price + dailyTrend + noise); data.push(price); } return data; } function drawChart(historical, projected) { const w = chartCanvas.width; const h = chartCanvas.height; ctx.clearRect(0, 0, w, h); const all = historical.concat(projected); const minP = Math.min(...all); const maxP = Math.max(...all); const pad = 20; function xScale(i, total) { return pad + (w - 2 * pad) * (i / (total - 1)); } function yScale(p) { return h - pad - (h - 2 * pad) * ((p - minP) / (maxP - minP || 1)); } // sfondo ctx.fillStyle = "#05070a"; ctx.fillRect(0, 0, w, h); // griglia semplice ctx.strokeStyle = "#111"; ctx.lineWidth = 1; for (let i = 0; i <= 4; i++) { const y = pad + (h - 2 * pad) * (i / 4); ctx.beginPath(); ctx.moveTo(pad, y); ctx.lineTo(w - pad, y); ctx.stroke(); } // storico (linea piena) ctx.strokeStyle = "#4fd18b"; ctx.lineWidth = 2; ctx.beginPath(); historical.forEach((p, i) => { const x = xScale(i, historical.length + projected.length); const y = yScale(p); if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); }); ctx.stroke(); // proiezione (linea tratteggiata) ctx.strokeStyle = "#f0c14b"; ctx.setLineDash([6, 4]); ctx.beginPath(); projected.forEach((p, i) => { const x = xScale(historical.length + i, historical.length + projected.length); const y = yScale(p); if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); }); ctx.stroke(); ctx.setLineDash([]); // legenda ctx.font = "11px system-ui"; ctx.fillStyle = "#ccc"; ctx.fillText("Storico (3 anni)", pad + 4, pad + 12); ctx.fillStyle = "#f0c14b"; ctx.fillText("Proiezione (1 mese)", pad + 4, pad + 26); } function computeEntryPrices(lastPrice, moneyFlow, liquidity) { // 5 prezzi di entrata attorno al prezzo attuale const entries = []; const baseVol = 0.04 + (1 - liquidity) * 0.04; // ampiezza for (let i = -2; i <= 2; i++) { const offset = i * baseVol * lastPrice; const price = lastPrice + offset; // probabilità sintetica: più alta se moneyFlow alto e prezzo leggermente scontato const discountFactor = 1 - Math.abs(i) * 0.08; let success = 40 + moneyFlow * 50 + discountFactor * 10; success = Math.max(10, Math.min(95, success)); entries.push({ price: price, success: success }); } // ordina per successo decrescente entries.sort((a, b) => b.success - a.success); return entries; } function successClass(p) { if (p >= 70) return "success-high"; if (p >= 50) return "success-mid"; return "success-low"; } function openDetail(inst, idx) { heatmapView.style.display = "none"; detailView.style.display = "block"; detailTitle.textContent = `${inst.ticker} – ${inst.name}`; // "Calcolo matematico complesso" sintetico basato su: // volumi (derivati da liquidity), domanda/offerta (moneyFlow), 3 anni di storico const historical = generateHistoricalSeries(inst.basePrice, inst.moneyFlow, inst.liquidity); const projected = projectFutureSeries(historical[historical.length - 1], inst.moneyFlow, inst.liquidity); drawChart(historical, projected); const avgPrice = historical.reduce((a, b) => a + b, 0) / historical.length; const volApprox = historical.reduce((acc, p) => acc + Math.abs(p - avgPrice), 0) / historical.length; const lastPrice = historical[historical.length - 1]; detailStats.innerHTML = `
Tipo strumento: ${inst.type}
Prezzo attuale sintetico: ${lastPrice.toFixed(2)}
Flusso domanda/offerta (0–1): ${inst.moneyFlow.toFixed(2)}
Liquidità (0–1): ${inst.liquidity.toFixed(2)}
Storico utilizzato: 3 anni (36 punti mensili sintetici)
Volatilità media sintetica: ${volApprox.toFixed(2)}
`; const entries = computeEntryPrices(lastPrice, inst.moneyFlow, inst.liquidity); entriesTableBody.innerHTML = ""; entries.forEach((e, i) => { const tr = document.createElement("tr"); const tdIdx = document.createElement("td"); const tdPrice = document.createElement("td"); const tdSucc = document.createElement("td"); tdIdx.textContent = i + 1; tdPrice.textContent = e.price.toFixed(2); tdSucc.textContent = e.success.toFixed(1) + " %"; tdSucc.className = successClass(e.success); tr.appendChild(tdIdx); tr.appendChild(tdPrice); tr.appendChild(tdSucc); entriesTableBody.appendChild(tr); }); } backBtn.addEventListener("click", () => { detailView.style.display = "none"; heatmapView.style.display = "block"; }); // Init renderHeatmap();
